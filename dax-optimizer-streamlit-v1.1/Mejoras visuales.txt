```python
from st_aggrid import AgGrid, GridOptionsBuilder, GridUpdateMode
from st_aggrid.shared import JsCode

def render_measures_table_aggrid(ranked_measures):
    # Preparar datos
    data = []
    for measure in ranked_measures:
        data.append({
            'Nombre': measure.name,
            'Tabla': measure.table,
            'Score': measure.impact_score,
            'Prioridad': measure.priority_label,
            'Cr√≠ticos': measure.critical_issues,
            'Warnings': measure.warnings,
            'Complejidad': measure.complexity
        })

    df = pd.DataFrame(data)

    # Configurar AG-Grid
    gb = GridOptionsBuilder.from_dataframe(df)
    gb.configure_default_column(
        resizable=True,
        filterable=True,
        sortable=True,
        editable=False
    )

    # Columna Score con color condicional
    cell_renderer = JsCode("""
    function(params) {
        if (params.value > 75) {
            return '<span style="color: white; background-color: #dc3545; padding: 5px 10px; border-radius: 12px; font-weight: bold;">' + params.value + '</span>';
        } else if (params.value > 50) {
            return '<span style="color: white; background-color: #fd7e14; padding: 5px 10px; border-radius: 12px; font-weight: bold;">' + params.value + '</span>';
        } else if (params.value > 25) {
            return '<span style="color: black; background-color: #ffc107; padding: 5px 10px; border-radius: 12px; font-weight: bold;">' + params.value + '</span>';
        } else {
            return '<span style="color: white; background-color: #28a745; padding: 5px 10px; border-radius: 12px; font-weight: bold;">' + params.value + '</span>';
        }
    }
    """)

    gb.configure_column("Score", cellRenderer=cell_renderer)

    # Configurar selecci√≥n de filas
    gb.configure_selection(selection_mode='multiple', use_checkbox=True)
    gb.configure_pagination(paginationAutoPageSize=True)

    gridOptions = gb.build()

    # Renderizar tabla
    AgGrid(
        df,
        gridOptions=gridOptions,
        update_mode=GridUpdateMode.SELECTION_CHANGED,
        height=600,
        theme='streamlit',  # Opciones: 'streamlit', 'alpine', 'balham', 'material'
        allow_unsafe_jscode=True
    )
```
